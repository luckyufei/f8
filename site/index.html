Easy To Use

npm install jmui
      
require(`jmui/styles/jmui.ios.core.css`)
require(`jmui/styles/jmui.ios.theme.${color}.css`)   /* optional */
import { Button } from 'jmui'

export default HelloWorld extends React.Component {
  render () {
    return <Button>Just do it!</Button>
  }
}
